#include<cstdio>
using namespace std;
int x,mark[1000],a[1000],cnt;
int main()
{ int n,t,i,flag;
 freopen("out.txt","w",stdout);
freopen("inp.txt","r",stdin);

    scanf("%d",&t);
    while(t--)
    { x++;
    for(i=0;i<1000;i++)
        mark[i]=0;

    cnt=0;
    flag=0;
        scanf("%d",&n);
        for(i=0;i<n;i++)
            scanf("%d",&a[i]);
        for(i=0;i<n;i++)
        {   if(a[i]==i)
            cnt+=1;
            mark[a[i]]+=1;
        }
        for(i=0;i<n;i++)
        {

            if(mark[i]>=2)
                flag=1;
        }
            if(flag==1)
            {
                printf("Case #%d: BAD\n",x);
            }
            else if(cnt>=2)
             printf("Case #%d: BAD\n",x);
            else
                 printf("Case #%d: GOOD\n",x);

    }
}
