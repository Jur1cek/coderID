#include <stdio.h>
#include<string.h>
int main(void) {
	char inp[200][50],out[200][50],a[50],b[50];
	int i,j,k,l,t,n,m,s,count,flag,prevcount,set;
	scanf("%d",&t);
	s=t+1;
	while(t--)
	{
		set=1;
		prevcount=40;
		scanf("%d %d",&n,&l);
		for(i=0;i<n;i++)
			scanf("%s",&inp[i][0]);
		
		for(i=0;i<n;i++)
			scanf("%s",&out[i][0]);
		
		for(i=0;i<n;i++)
		{
			for(j=0;j<l;j++)
			{
				if(inp[i][j]!=out[0][j])
					a[j]='1';
				else
					a[j]='0';
			}
			a[j]=NULL;
			for(k=0;k<n;k++)
			{
				for(j=0;j<l;j++)
				{
					if(out[k][j]!=a[j])
						b[j]='1';
					else
						b[j]='0';
				}
				b[j]=NULL;
				flag=1;
				for(m=0;m<n;m++)
				{
					if(!strcmp(inp[m],b))
						flag=2;
				}
				if(flag==1)
					break;
			}
			if(k==n && flag==2)
			{
				count=0;
				for(m=0;m<l;m++)
				{
					if(a[m]=='1')
						count++;
				}
				set=2;
				if(count>prevcount)
					count=prevcount;
				prevcount=count;
			}
		}
		if(set!=2)
			printf("Case #%d: NOT POSSIBLE\n",s-t-1);
		else
			printf("Case #%d: %d\n",s-t-1,count);
	}
	return 0;
}
